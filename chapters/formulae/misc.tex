The reflection of point $\vec{Q}$ w.r.t a line is given by
\begin{align}
	\label{eq:11/10/4/22}
\vec{R} = \vec{Q} -\frac{2\brak{\vec{n}^{\top}\vec{Q}-c}}{\norm{\vec{n}}}\vec{n}
\end{align}
\end{enumerate}
